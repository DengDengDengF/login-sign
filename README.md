### 2.依赖说明

```js
    "@expo/metro-runtime": "~3.2.1",//启动开发服务器并在浏览器中进行开发
    "@react-navigation/native": "^6.1.17",//React Navigation库的核心包，用于在React Native应用中实现导航功能。它提供了导航容器和基本的导航功能
    "@react-navigation/native-stack": "^6.9.26",//用于实现基于堆栈的导航，这种导航方式类似于网页的前进和后退操作。它是React Navigation的一个具体实现，专门用于堆栈导航。
    "expo": "~51.0.9",//Expo SDK的核心包，包含了许多开箱即用的API和工具，简化了React Native应用的开发过程。
    "expo-status-bar": "~1.12.1",//用于控制和管理应用的状态栏。它提供了简单的方法来设置状态栏的样式和显示状态。
    "react": "18.2.0",//React库的核心包，提供了用于构建用户界面的基本组件和工具。
    "react-dom": "18.2.0",//React的DOM渲染包，用于将React组件渲染到网页的DOM中。虽然在React Native项目中主要用于Web支持。
    "react-native": "0.74.1",//React Native库的核心包，提供了构建跨平台移动应用的基本组件和工具。
    "react-native-date-picker": "^5.0.4",//一个用于选择日期的React Native组件，提供了直观的日期选择界面。
    "react-native-safe-area-context": "4.10.1",//用于处理iOS和Android设备的安全区域（Safe Area），确保应用内容不会被设备的物理特性（如刘海屏）遮挡。
    "react-native-screens": "3.31.1",//优化的屏幕组件，帮助提高导航性能，特别是在使用React Navigation时。
    "react-native-web": "~0.19.10",//将React Native组件和API映射到Web平台，使得可以使用相同的代码在Web上运行React Native应用。
    "@react-native-community/datetimepicker": "8.0.1"//一个跨平台的日期和时间选择器，提供了原生的日期和时间选择功能。
```
